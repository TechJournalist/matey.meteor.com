    Takes a Board, returns a new one on which the move has been played.

      move: (board, from, to) ->
    newPieces = _.clone(board.pieces)
    mover = newPieces[from]
    delete newPieces[from]
    newPieces[to] = mover
    {
      pieces: newPieces
    }
